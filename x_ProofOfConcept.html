// ── Reactive hash ─────────────────────────────────────────────────────────────
// Re-evaluates automatically whenever momScores changes.
mom_hash = {
  momScores;   // reactive dependency

  const scores = momScores;

  // Build per-question summary objects
  const questions = {};
  for (const [descriptor, scoreString] of Object.entries(scores)) {
    if (scoreString === "unattempted") {
      questions[descriptor] = {
        raw_scores: "unattempted",
        parts_total: null,
        parts_correct: null,
        score: null,
        result: "unattempted"
      };
    } else {
      const parts = scoreString.includes("~")
        ? scoreString.split("~").map(Number)
        : [parseFloat(scoreString)];
      const total = scoreString.includes("~") ? parts.length : 1;
      const correct = parts.reduce((a, b) => a + b, 0);
      const score = correct / total;
    
      questions[descriptor] = {
        raw_scores: scoreString,
        parts_total: total,
        parts_correct: correct,
        score: score,
        result: score === 1 ? "fully_correct" : score === 0 ? "incorrect" : "partially_correct"
      };
    }
  }

  const payload = {
    notebook: "Proof of Concept: Embedded MyOpenMath",
    type: "mom_questions",
    timestamp: new Date().toISOString(),
    questions: questions
  };

  return toBase64(JSON.stringify(payload));
}
