problemRegistry = ({
  "q1_combine_like_terms": "q1_combine_like_terms",
  "q2_simplify_rational":  "q2_simplify_rational",
  "q3_match_trig":         "q3_match_trig"
})
mutable momScores = Object.fromEntries(
  Object.values(problemRegistry).map(d => [d, "unattempted"])
)
{
  window.addEventListener("message", function(event) {
    if (event.origin !== "https://www.myopenmath.com") return;

    let data;
    try {
      data = typeof event.data === "string" ? JSON.parse(event.data) : event.data;
    } catch(e) { return; }

    if (data.action === "updatescores" && Array.isArray(data.scores)) {
      try {
        const jwtPayload = JSON.parse(atob(data.jwt.split(".")[1]));
        const frameId = jwtPayload.targetid;
        const descriptor = problemRegistry[frameId];
        if (descriptor && data.scores.length > 0) {
          const current = mutable momScores;
          const updated = Object.assign({}, current);
          const raw = data.scores[0];
          updated[descriptor] = typeof raw === "number" ? String(raw) : raw;
          mutable momScores = updated;
        }
      } catch(e) { return; }
    }
  });
}
function toBase64(str) {
  return btoa(unescape(encodeURIComponent(str)));
}

mom_hash = {
  momScores;

  const scores = momScores;
  const questions = {};

  for (const [descriptor, scoreString] of Object.entries(scores)) {
    if (scoreString === "unattempted") {
      questions[descriptor] = {
        raw_scores: "unattempted",
        parts_total: null,
        parts_correct: null,
        score: null,
        result: "unattempted"
      };
    } else {
      const s = String(scoreString);
      const parts = s.includes("~")
        ? s.split("~").map(Number)
        : [parseFloat(s)];
      const total = parts.length;
      const correct = parts.reduce((a, b) => a + b, 0);
      const score = correct / total;
      questions[descriptor] = {
        raw_scores: s,
        parts_total: total,
        parts_correct: correct,
        score: score,
        result: score === 1 ? "fully_correct" : score === 0 ? "incorrect" : "partially_correct"
      };
    }
  }

  const payload = {
    notebook: "Proof of Concept: Embedded MyOpenMath",
    type: "mom_questions",
    timestamp: new Date().toISOString(),
    questions: questions
  };

  return toBase64(JSON.stringify(payload));
}
